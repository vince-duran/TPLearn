üíª Code Changes

1. Error Handler (catch block)

// BEFORE (Context could be lost)
const isAssessment = error.isAssessment !== undefined 
    ? error.isAssessment 
    : (context === 'assessment' ? true : false);

// AFTER (Context is primary)
let isAssessment;
if (context === 'assessment') {
    isAssessment = true; // ALWAYS true for assessments
} else if (context === 'assignment') {
    isAssessment = false; // ALWAYS false for assignments
} else {
    isAssessment = error.isAssessment ?? null; // Fallback
}

2. Empty State Message

// BEFORE (Hardcoded "assignment")
tableBody.innerHTML = '...No submissions found for this assignment.';

// AFTER (Context-aware)
const context = window.currentSubmissionContext;
const itemType = context === 'assessment' ? 'assessment' 
              : context === 'assignment' ? 'assignment' 
              : 'item';
tableBody.innerHTML = `...No submissions found for this ${itemType}.`;
